```typescript
import { NextResponse } from 'next/server';
import type { NextRequest } from 'next/server';

// -- Rate Limiting (in-memory token bucket) --

interface RateLimitEntry {
  tokens: number;
  lastRefill: number;
}

const rateLimitMap = new Map<string, RateLimitEntry>();
const RATE_LIMIT_MAX_MAP_SIZE = 10000;  // Prevent memory leak (Pitfall 4)

// Per-path rate limit configuration (per D-05: cost APIs only)
const RATE_LIMITS: Record<string, { maxTokens: number; refillIntervalMs: number }> = {
  '/api/ai-insight': { maxTokens: 10, refillIntervalMs: 60_000 },   // 10 req/min
  '/api/transcript': { maxTokens: 20, refillIntervalMs: 60_000 },   // 20 req/min
  '/api/news': { maxTokens: 20, refillIntervalMs: 60_000 },         // 20 req/min
};

function checkRateLimit(ip: string, pathname: string): boolean {
  const config = Object.entries(RATE_LIMITS).find(([path]) => pathname.startsWith(path));
  if (!config) return true;  // Not rate-limited path

  const [, { maxTokens, refillIntervalMs }] = config;
  const key = `${ip}:${pathname}`;
  const now = Date.now();

  // Cleanup stale entries to prevent memory leak (Pitfall 4)
  if (rateLimitMap.size > RATE_LIMIT_MAX_MAP_SIZE) {
    const cutoff = now - refillIntervalMs * 10;
    for (const [k, v] of rateLimitMap) {
      if (v.lastRefill < cutoff) rateLimitMap.delete(k);
    }
  }

  const entry = rateLimitMap.get(key) ?? { tokens: maxTokens, lastRefill: now };
  const elapsed = now - entry.lastRefill;
  const refills = Math.floor(elapsed / refillIntervalMs);
  if (refills > 0) {
    entry.tokens = Math.min(maxTokens, entry.tokens + refills * maxTokens);
    entry.lastRefill = now;
  }
  if (entry.tokens <= 0) return false;
  entry.tokens--;
  rateLimitMap.set(key, entry);
  return true;
}

// -- Basic Auth --

function checkBasicAuth(request: NextRequest): boolean {
  const user = process.env.DASHBOARD_USER;
  const pass = process.env.DASHBOARD_PASS;
  if (!user || !pass) return true;  // Skip auth if not configured (dev mode)

  const authHeader = request.headers.get('authorization');
  if (!authHeader?.startsWith('Basic ')) return false;

  try {
    const decoded = atob(authHeader.slice(6));
    const [u, p] = decoded.split(':');
    return u === user && p === pass;
  } catch {
    return false;
  }
}

// -- Proxy Function (Next.js 16.1 convention) --

export function proxy(request: NextRequest) {
  const { pathname } = request.nextUrl;

  // Basic Auth check for all API routes (per D-04)
  if (!checkBasicAuth(request)) {
    return new NextResponse('Authentication required', {
      status: 401,
      headers: { 'WWW-Authenticate': 'Basic realm="Dashboard"' },
    });
  }

  // Rate limiting for expensive API routes only (per D-05)
  const ip = request.headers.get('x-forwarded-for')?.split(',')[0]?.trim() ?? 'unknown';
  if (!checkRateLimit(ip, pathname)) {
    return NextResponse.json(
      { error: 'Too many requests' },
      { status: 429, headers: { 'Retry-After': '60' } },
    );
  }

  return NextResponse.next();
}

export const config = {
  matcher: '/api/:path*',
};
```

**Key design decisions:**
- Export name is `proxy` not `middleware` (Next.js 16.1 convention)
- File location is `src/proxy.ts` not `src/middleware.ts`
- Auth skipped when env vars not set (dev convenience -- per D-04 "Vercel 대시보드에서 환경변수만 설정하면 동작")
- Rate limit per path+IP combination, not just IP
- Rate limit config per endpoint: 10/min for ai-insight (most expensive), 20/min for transcript/news
- Map size capped at 10000 entries with stale cleanup (Pitfall 4 prevention)
- Data-query APIs (`/api/supply-chain`, `/api/vcm`, `/api/customer-detail`) pass through rate limiting (per D-05 "데이터 조회 API는 DB만 접근하므로 불필요")
- `atob` decoding in try/catch for malformed base64 headers
